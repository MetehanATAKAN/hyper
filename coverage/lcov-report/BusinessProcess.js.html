import React, { useState } from 'react';
import { Button, Form } from 'react-bootstrap';
import Select from 'react-select';
import { FetchApiPost } from '../../../utils/http.helper';
 
const BusinessProcess = ({ setOnModal, businessProcess }) => {
    const [selectActivity, setSelectActivity] = useState({
        id: 0,
        value: 'select activity types',
        label: 'select activity types',
    });
    const [selectJobDescription, setSelectJobDescription] = useState({
        id: 0,
        value: 'select job description',
        label: 'select job description',
    });
    const [businessProcessName, setBusinessProcessName] = useState('');
    const handleChange = (e) => {
        setBusinessProcessName(e.target.value);
    };
    const loggedUserName = localStorage.getItem('userName');
    const handleSubmit = (e) => {
        e.preventDefault();
        if (businessProcessName !== '') {
            (async () => {
                try {
                    const businessData = await {
                        businesProcessTitle: businessProcessName,
                        createdBy: loggedUserName,
                        jobDescriptionId: Number(selectJobDescription.id),
                        category: 'A',
                        ownerBP: loggedUserName,
                    };
                    await FetchApiPost(
                        'services/TaskManagement/BusinessProcess/CreateBusinessProcess',
                        'POST',
                        businessData
                    );
                    setBusinessProcessName('');
                    setOnModal(false);
                } catch (error) {
                    console.log(error);
                }
            })();
        } else {
            setOnModal(false);
        }
    };
    const changeActivity = (e) => {
        setSelectActivity(e);
    };
    const changeJobDescription = (e) => {
        setSelectJobDescription(e);
    };
 
    return (
        <>
            <div className="job-description-container">
                <div className="parent-process-activity-type">
                    <label>activity types</label>
                    <Select
                        isMulti={false}
                        className="react-select"
                        placeholder="select"
                        classNamePrefix="react-select"
                        value={selectActivity}
                        onChange={changeActivity}
                        options={businessProcess[0]?.map((option) => ({
                            id: option.activityTypeId,
                            value: option.activityTypeName,
                            label: option.activityTypeName,
                        }))}></Select>
                </div>
                <div className="parent-process-job-description">
                    <label>job description</label>
                    <Select
                        isDisabled={selectActivity.id === 0}
                        isMulti={false}
                        className="react-select"
                        placeholder="select"
                        classNamePrefix="react-select"
                        onChange={changeJobDescription}
                        value={selectJobDescription}
                        options={businessProcess[0]?.map((option) => ({
                            id: option.mainProcessId,
                            value: option.mainProcessName,
                            label: option.mainProcessName,
                        }))}></Select>
                </div>
 
                <div className="business-process-process-select">
                    <label>add business process</label>
                    <Form.Control
                        disabled={selectJobDescription.id === 0}
                        type="text"
                        data-testid="business-process-name"
                        value={businessProcessName}
                        onChange={handleChange}
                        placeholder="add business process"></Form.Control>
                </div>
            </div>
            <div className="task-management-footer-btn">
                <Button variant="danger" onClick={() => setOnModal(false)}>
                    cancel
                </Button>
                <Button data-testid="create-business-process" variant="primary" onClick={handleSubmit}>
                    add
                </Button>
            </div>
        </>
    );
};
 
export default BusinessProcess;
